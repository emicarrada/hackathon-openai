\begin{block}{Proceso Self-Refine}
\begin{enumerate}
    \item \textbf{Generaci√≥n Inicial ($y_0$):} LLM produce salida inicial
    \item \textbf{Feedback ($fb_t$):} El mismo LLM critica su propia salida
    \item \textbf{Refine ($y_{t+1}$):} LLM usa feedback para mejorar
    \item \textbf{Repetir:} Hasta cumplir condici√≥n de parada
\end{enumerate}
\end{block}
\end{frame}
